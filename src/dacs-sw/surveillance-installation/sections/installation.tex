% Procedure for installation

\stepcounter{tableCounter} % Increment counter
\setcounter{rowCounter}{0} % Reset counter
\begin{tabularx}{\textwidth}{|>{\columncolor{tableColumnColor}}c|>{\columncolor{tableColumnColor}}c|>{\columncolor{tableColumnColor}}c|>{\columncolor{tableColumnColor}}c|X|}
  \hline
  \rowcolor{tableHeaderColor}
  ID & CK 1 & CK 2 & CK 3 & Description \\ \hline

  \procedureItem{
    Check that the surveillance cameras are positioned on the trailer:
    \begin{itemize}
      \item 2 cameras on the FSS side
      \item 2 cameras on the OSS side
      \item 2 cameras on the shielding wall, on the engine side
    \end{itemize}
    For range surveillance, 1 camera will be mounted on the earth wall.
  }

  \procedureItem{
    Check inside the junction box on the OSS tower, that every power camera is properly connected
  }

  \procedureItem{
    Make sure the Modem is connected via cable to the lan network of the trailer
  }

  \procedureItem{
    Connect ethernet cable from surveillance.
  }

  \procedureItem{
    After Mission Control Station is installed as in start surveillance website via the shortcut in the bottom task bar
  }

  \procedureItem{
    Login via the button (no password needed, username = admin)
  }

  \procedureItem{
    \sout{
      Go to settings --> Video settings
    }
  }

  \procedureItem{
    \sout{
      Mark all “Timing” Boxes red of the weekday the test is taking place.
      This will lead to the surveillance system recording said timeframe.
    }
  }

  \procedureItem{
    \sout{
      On the bottom select “Select all” and press save.
      All Cameras should be saved continuously now.
      Recordings will be saved on the surveillance box and have to be downloaded after each firing day by connecting a screen to the surveillance box
    }
  }

  \procedureItem{
    Check that cameras see all points of interest (valves, displays, etc.)
  }


  \procedureItem{
    To record the video Press Windows-G and then Record
  }


  \procedureItem{
    Turn off the DACS box after the test day is over
  }


  \procedureItem{
    Disconnect monitor and modem and prepare the modem for its installation on the trailer
  }
\end{tabularx}
